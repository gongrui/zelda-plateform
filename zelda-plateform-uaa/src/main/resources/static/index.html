Sa-Token OAuth2 PKCE Demo

什么是 PKCE？

PKCE（Proof Key for Code Exchange）是一种增强 OAuth 2.0 安全性的扩展协议，主要用于公共客户端（如移动应用、单页应用），防止授权码拦截攻击。

核心流程：

  1. 客户端生成随机 code_verifier
  2. 计算 code_challenge = BASE64URL(SHA256(code_verifier))
  3. 授权请求携带 code_challenge 和 code_challenge_method
  4. 令牌请求携带原始 code_verifier 供服务器验证

配置说明

// PKCE 参数生成示例 (JavaScript)

function base64UrlEncode(buffer) {
    return btoa(String.fromCharCode.apply(null, new Uint8Array(buffer)))
        .replace(/\+/g, '-')
        .replace(/\//g, '_')
        .replace(/=/g, '');
}

function generatePkceParams() {
    // 1. 生成 code_verifier (43-128个字符)
    const codeVerifier = base64UrlEncode(crypto.getRandomValues(new Uint8Array(32)));
    
    // 2. 使用 SHA256 计算 code_challenge
    crypto.subtle.digest('SHA-256', new TextEncoder().encode(codeVerifier))
        .then(buffer => {
            const codeChallenge = base64UrlEncode(buffer);
            console.log('code_verifier:', codeVerifier);
            console.log('code_challenge:', codeChallenge);
        });
}

API 端点

// 授权端点
GET /oauth2/authorize
  ?response_type=code
  &client_id=your_client_id
  &redirect_uri=your_redirect_uri
  &scope=user_info
  &state=random_state
  &code_challenge=your_code_challenge
  &code_challenge_method=S256

// 令牌端点
POST /oauth2/token
Content-Type: application/x-www-form-urlencoded

grant_type=authorization_code
&client_id=your_client_id
&client_secret=your_client_secret
&code=authorization_code
&redirect_uri=your_redirect_uri
&code_verifier=your_code_verifier
